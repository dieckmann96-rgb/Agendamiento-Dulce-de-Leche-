    <script>
        // --- CONFIGURAÇÕES GLOBAIS ---
        const WHATSAPP_NUMBER = "595982644359"; // O número de WhatsApp sem o '+'
        const WHATSAPP_CODE = "+595"; // Código de área para exibição
        
        // Regras de Horário de Funcionamento
        const BUSINESS_HOURS = {
            WEEKDAY: { start: 15, end: 22 }, // Segunda a Sexta: 15:00 a 22:00
            WEEKEND: { start: 9, end: 20 }   // Sábado e Domingo: 09:00 às 20:00
        };

        // Lista de Serviços (Organizados e Listados)
        const SERVICES = [
            "Manicure e Pedicure - Normal",
            "Manicure e Pedicure - Semi-permanente",
            "Extensão de Cílios - Carga Leve",
            "Extensão de Cílios - Mediano",
            "Extensão de Cílios - Carga Total",
            "Design de Sobrancelhas - Com Henna",
            "Design de Sobrancelhas - Com Tinta",
            "Maquiagem Profissional - Com Cílios",
            "Maquiagem Profissional - Sem Cílios",
            "Tratamento Capilar - Lavado e Secado",
            "Tratamento Capilar - Com Hidratação",
            "Laminado de Sobrancelhas",
            "Lifting de Cílios",
        ];

        // --- REFERÊNCIAS DE ELEMENTOS DO DOM ---
        const form = document.getElementById('scheduling-form');
        const serviceSelect = document.getElementById('service');
        const dateInput = document.getElementById('date');
        const timeSelect = document.getElementById('time');
        const nameInput = document.getElementById('name');
        const phoneInput = document.getElementById('phone');
        const hoursInfo = document.getElementById('hours-info');
        const messageBox = document.getElementById('message-box');

        // --- FUNÇÕES DE UTILIDADE ---

        /**
         * Exibe uma mensagem de notificação (Erro ou Sucesso).
         * @param {string} message - A mensagem a ser exibida.
         * @param {string} type - 'success' ou 'error'.
         */
        function showMessage(message, type) {
            messageBox.textContent = message;
            messageBox.className = 'mt-4 p-3 rounded-lg text-sm transition-all duration-300 border';
            messageBox.classList.remove('hidden');

            if (type === 'error') {
                messageBox.classList.add('bg-red-100', 'text-red-700', 'border-red-300');
            } else {
                messageBox.classList.add('bg-green-100', 'text-green-700', 'border-green-300');
            }
            setTimeout(() => {
                messageBox.classList.add('hidden');
            }, 6000);
        }

        /**
         * Gera slots de 30 em 30 minutos dentro de um intervalo.
         * @param {number} startHour - Hora de início (0-23).
         * @param {number} endHour - Hora de fim (0-23).
         * @returns {string[]} - Array de strings de horário (e.g., "15:00").
         */
        function generateTimeSlots(startHour, endHour) {
            const slots = [];
            let currentHour = startHour;
            let currentMinute = 0;

            // O loop deve ir até a hora anterior ao fim (pois o fim é o limite, o último agendamento deve começar 30 minutos antes do 'end')
            while (currentHour * 60 + currentMinute < endHour * 60) {
                const hourStr = String(currentHour).padStart(2, '0');
                const minuteStr = String(currentMinute).padStart(2, '0');
                slots.push(`${hourStr}:${minuteStr}`);

                // Avança 30 minutos
                currentMinute += 30;
                if (currentMinute === 60) {
                    currentMinute = 0;
                    currentHour++;
                }
            }
            return slots;
        }

        /**
         * Preenche o campo de Horário baseado na data selecionada.
         */
        function populateTimeSlots() {
            const selectedDateValue = dateInput.value;
            timeSelect.innerHTML = '<option value="" disabled selected>Selecione um Horário</option>';
            timeSelect.disabled = true;
            timeSelect.classList.add('bg-gray-100', 'cursor-not-allowed');
            timeSelect.classList.remove('bg-white', 'cursor-pointer');
            hoursInfo.classList.add('hidden');

            if (!selectedDateValue) return;

            // A data nativa do input é no formato YYYY-MM-DD
            const selectedDate = new Date(selectedDateValue + 'T00:00:00'); 
            const dayOfWeek = selectedDate.getDay(); // 0 (Domingo) a 6 (Sábado)

            let rules;
            let rulesText;

            // 1-5 é Segunda a Sexta (WEEKDAY)
            if (dayOfWeek >= 1 && dayOfWeek <= 5) {
                rules = BUSINESS_HOURS.WEEKDAY;
                rulesText = `Horários: Segunda a Sexta, das ${rules.start}:00 às ${rules.end}:00`;
            } 
            // 0 é Domingo, 6 é Sábado (WEEKEND)
            else if (dayOfWeek === 0 || dayOfWeek === 6) {
                rules = BUSINESS_HOURS.WEEKEND;
                rulesText = `Horários: Sábado e Domingo, das ${rules.start}:00 às ${rules.end}:00`;
            } else {
                return;
            }

            const slots = generateTimeSlots(rules.start, rules.end);

            if (slots.length === 0) {
                timeSelect.innerHTML = '<option value="" disabled selected>Nenhum horário disponível neste dia.</option>';
                hoursInfo.textContent = 'Não há horários disponíveis para agendamento.';
                hoursInfo.classList.remove('hidden');
            } else {
                slots.forEach(slot => {
                    const option = document.createElement('option');
                    option.value = slot;
                    option.textContent = slot;
                    timeSelect.appendChild(option);
                });
                
                timeSelect.disabled = false;
                timeSelect.classList.remove('bg-gray-100', 'cursor-not-allowed');
                timeSelect.classList.add('bg-white', 'cursor-pointer');
                hoursInfo.textContent = rulesText;
                hoursInfo.classList.remove('hidden');
            }
        }

        /**
         * Inicializa o formulário: Preenche serviços e configura a data mínima.
         */
        function initializeForm() {
            // 1. Preenche o Dropdown de Serviços
            SERVICES.forEach(service => {
                const option = document.createElement('option');
                option.value = service;
                option.textContent = service;
                serviceSelect.appendChild(option);
            });

            // 2. Configura a Data Mínima (apenas hoje ou datas futuras)
            const today = new Date();
            const year = today.getFullYear();
            const month = String(today.getMonth() + 1).padStart(2, '0');
            const day = String(today.getDate()).padStart(2, '0');
            dateInput.min = `${year}-${month}-${day}`;
            
            // 3. Adiciona Listeners
            dateInput.addEventListener('change', populateTimeSlots);
            form.addEventListener('submit', handleFormSubmit);
        }
        
        /**
         * Manipula o envio do formulário, criando e abrindo o link do WhatsApp.
         * @param {Event} e - O evento de submissão do formulário.
         */
        function handleFormSubmit(e) {
            e.preventDefault();

            const service = serviceSelect.value;
            const date = dateInput.value;
            const time = timeSelect.value;
            const name = nameInput.value.trim();
            const phone = phoneInput.value.trim();

            if (!service || !date || !time || !name || !phone) {
                showMessage("Por favor, preencha todos os campos do agendamento.", 'error');
                return;
            }

            // Formatação da mensagem
            const message = `Olá! Gostaria de agendar um horário.
            
*Detalhes do Agendamento:*
*Nome:* ${name}
*Telefone:* ${WHATSAPP_CODE} ${phone}
*Serviço:* ${service}
*Data:* ${date.split('-').reverse().join('/')}
*Horário:* ${time}
            
Por favor, confirme a disponibilidade e o valor total do serviço. Obrigado(a)!`;
            
            // Codifica a mensagem para o URL
            const encodedMessage = encodeURIComponent(message);
            
            // Cria o link do WhatsApp
            const whatsappUrl = `https://api.whatsapp.com/send?phone=${WHATSAPP_NUMBER}&text=${encodedMessage}`;

            // Abre o link em uma nova aba
            window.open(whatsappUrl, '_blank');

            showMessage("Agendamento enviado! Redirecionando para o WhatsApp para confirmação...", 'success');
            
            // Opcional: Limpar o formulário após o envio
            // form.reset();
            // populateTimeSlots(); 
        }

        // Inicia a aplicação
        initializeForm();
        
    </script>
